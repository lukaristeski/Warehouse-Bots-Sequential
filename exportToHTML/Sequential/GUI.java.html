package Sequential;


import javax.imageio.ImageIO;
import javax.swing.*;
import java.awt.*;
import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;

public class GUI extends JPanel implements Runnable {

    private static final int FRAME_WIDTH = 800;
    private static final int FRAME_HEIGHT = 600;

    private final BufferedImage floor = ImageIO.read(new File("./Tile/tile_0468.png"));
    private final BufferedImage botImage = ImageIO.read(new File("./Tiles/tile_0108.png"));
    private final BufferedImage shelfImage = ImageIO.read(new File("./Tile/tile_0300.png"));


    public GUI() throws IOException {
        JFrame frame = new JFrame("Warehouse Bots");
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setSize(FRAME_WIDTH, FRAME_HEIGHT + 39);
        frame.add(this);
        frame.setVisible(true);
    }

    @Override
    public void run() {
        Timer timer = new Timer(200, e -> {
            for (Bot bot : GameState.bots){
                bot.update();
            }
            repaint();
        });
        timer.start();
    }

    @Override
    public void paint(Graphics g) {
        Graphics2D g2d = (Graphics2D) g;

        int tileWidth = FRAME_WIDTH / GameState.WIDTH;
        int tileHeight = FRAME_HEIGHT / GameState.HEIGHT;

        for (int row = 0; row < GameState.HEIGHT; row++) {
            for (int col = 0; col < GameState.WIDTH; col++) {
                int x = col * tileWidth;
                int y = row * tileHeight;
                g2d.drawImage(floor, x, y, tileWidth, tileHeight, null);
            }
        }

        for (Shelf shelf : GameState.shelves) {
            Location loc = shelf.getLocation();
            int x = loc.getX() * tileWidth;
            int y = loc.getY() * tileHeight;
            g2d.drawImage(shelfImage, x, y, tileWidth, tileHeight, null);
        }

        for (Bot bot : GameState.bots) {
            Location loc = bot.getLocation();
            int botX = loc.getX() * tileWidth;
            int botY = loc.getY() * tileHeight;
            g2d.drawImage(botImage, botX, botY, tileWidth, tileHeight, null);
        }
    }
}
