package Sequential;

import java.util.LinkedList;
import java.util.Queue;

public class Bot {

    public Location location;
    private Queue<Task> tasks;

    private int dir = 0;

    private Integer targetX = null;
    private Integer targetY = null;

    public Bot(Location start) {
        this.location = start;
        this.tasks = new LinkedList<>();
    }

    public Location getLocation(){
        return location;
    }

    public void addTask(Task task) {
        tasks.add(task);
    }

    public Task getCurrentTask() {
        return tasks.peek();
    }

    public void finishCurrentTask() {
        tasks.poll();
    }

    public boolean hasTasks() {
        return !tasks.isEmpty();
    }

    public void setTarget(int x, int y) {
        targetX = x;
        targetY = y;
    }

    public void clearTarget(){
        targetX = null;
        targetY = null;
    }

    public boolean hasTarget(){
        return targetX != null && targetY != null;
    }

    public boolean atTarget(){
        return hasTarget() &&
                location.getX() == targetX &&
                location.getY() == targetY;
    }

    public void step(){
        int[][] dirs = {
                {1,0},
                {0,1},
                {-1,0},
                {0,-1},
        };
        if (tryMove(dirs[dir][0],dirs[dir][1])){
            return;
        }
        for (int i = 0; i < 4; i++){
            dir = (dir + 1) % 4;
            if (tryMove(dirs[dir][0],dirs[dir][1])){
                return;
            }
        }
    }

    public void moveTowardTarget(){
        if (!hasTarget()){
            return;
        }
        int dx = Integer.compare(targetX, location.getX());
        int dy = Integer.compare(targetY, location.getY());

        if (dx != 0 && tryMove(dx,0)){
            return;
        }
        if (dy != 0 && tryMove(dy,0)){
            return;
        }
        if (tryMove(0,1)){
            return;
        }
        if (tryMove(1,0)){
            return;
        }
        if (tryMove(0,-1)){
            return;
        }
        tryMove(1,-1);

    }

    public boolean tryMove(int dx, int dy) {
        int nx = location.getX() + dx;
        int ny = location.getY() + dy;

        System.out.println("trying move to " + nx + "," + ny);

        if (GameState.isWalkable(nx, ny)) {
            location.setX(nx);
            location.setY(ny);
            System.out.println("moved");
            return true;
        }
        System.out.println("blocked");
        return false;
    }

    public void update(){
        if (!hasTasks()){

            Task task = getCurrentTask();

            int targetShelfX;
            int targetShelfY;
            int slot;

            if (task.getStage() == Task.Stage.PICKUP){
                targetShelfX = task.getPickupX();
                targetShelfY = task.getPickupY();
                slot = task.getPickupSlot();
            } else if (task.getStage() == Task.Stage.DROPOFF) {
                targetShelfX = task.getDropX();
                targetShelfY = task.getDropY();
                slot = task.getDropSlot();
            } else {
                finishCurrentTask();
                return;
            }

            Location target = GameState.getInteractionTile(targetShelfX, targetShelfY);
            if (target == null) return;

            if (location.getX() == target.getX() && location.getY() == target.getY()){

                Shelf shelf = GameState.getShelByShelfGrid(targetShelfX,targetShelfY);
                if (shelf == null) return;

                boolean ok;
                if (task.getStage() == Task.Stage.PICKUP){
                    ok = shelf.pick(slot);
                    if (ok) task.advance();
                }else {
                    ok = shelf.drop(slot);
                    if (ok) task.advance();
                    if (task.isDone()) finishCurrentTask();
                }
                return;
            }

            int dx = Integer.compare(target.getX(), location.getX());
            int dy = Integer.compare(target.getY(), location.getY());

            int nextX = location.getX() + dx;
            int nextY = location.getY() + dy;

            if (dx != 0 && !GameState.isOccupied(nextX,nextY)){
                location.setX(nextX);
                return;
            }

            nextX = location.getX();
            nextY = location.getY() + dy;

            if (dy != 0 && !GameState.isOccupied(nextX,nextY)){
                location.setY(nextY);
            }

        }
    }

}
