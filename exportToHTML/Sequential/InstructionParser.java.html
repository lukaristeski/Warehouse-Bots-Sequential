package Sequential;

import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.util.ArrayList;

public class InstructionParser {

    public static void parse(String filename) throws IOException {
        BufferedReader reader = new BufferedReader(new FileReader(filename));

        int numBots = Integer.parseInt(reader.readLine().trim());

        GameState.bots = new ArrayList<>();

        for (int i = 0; i < numBots; i++) {
            GameState.bots.add(new Bot(GameState.findFreeTile()));
        }

        String[] parts =  reader.readLine().split(" ");
        int shelfCols = Integer.parseInt(parts[0]);
        int shelfRows = Integer.parseInt(parts[1]);

        GameState.WIDTH = shelfCols + (shelfCols - 1);
        GameState.HEIGHT = shelfRows + (shelfRows - 1);

        GameState.initShelves(shelfCols, shelfRows);

        String line;
        while ((line = reader.readLine()) != null) {
            line = line.trim();
            if (line.isEmpty()) {
                continue;
            }

            if (!line.contains("|") || !line.contains("(from)") || !line.contains("(to)")) {
                System.err.println("Skipping non-task line: " + line);
                continue;
            }

            String[] taskParts = line.split("\\|");

            if (taskParts.length != 3) {
                System.err.println("Invalid task line: " + line);
                continue;
            }

            int botIndex = Integer.parseInt(taskParts[0].substring(1)) - 1;

            String from = taskParts[1].replace("(from)","").replace(" ","");
            String[] f = from.split("-");
            if (f.length != 3) {
                System.err.println("Invalid from: " + taskParts[1]);
                continue;
            }
            int fromX = Integer.parseInt(f[0]);
            int fromY = Integer.parseInt(f[1]);
            int fromSlot = Integer.parseInt(f[2],16);

            String to = taskParts[2].replace("(to)", "").replace(" ","");
            String[] t = to.split("-");
            if (t.length != 3) {
                System.err.println("Invalid to: " + taskParts[2]);
                continue;
            }
            int toX = Integer.parseInt(t[0]);
            int toY = Integer.parseInt(t[1]);
            int toSlot = Integer.parseInt(t[2],16);

            Task task = new Task(fromX,fromY,fromSlot,toX,toY,toSlot);

            GameState.getBot(botIndex).addTask(task);
        }
        reader.close();
    }
}
