package Sequential;

public class Task {

    public enum Stage {
        PICKUP,
        DROPOFF,
        DONE,
    }

    private final int pickupX;
    private final int pickupY;
    private final int pickupSlot;

    private final int dropX;
    private final int dropY;
    private final int dropSlot;

    private Stage stage;

    public Task(int pickupX, int pickupY, int pickupSlot, int dropX, int dropY, int dropSlot) {

        this.pickupX = pickupX;
        this.pickupY = pickupY;
        this.pickupSlot = pickupSlot;

        this.dropX = dropX;
        this.dropY = dropY;
        this.dropSlot = dropSlot;

        this.stage = Stage.PICKUP;
    }


    public int getPickupX() {
        return pickupX;
    }

    public int getPickupY() {
        return pickupY;
    }

    public int getPickupSlot() {
        return pickupSlot;
    }

    public int getDropX() {
        return dropX;
    }

    public int getDropY() {
        return dropY;
    }

    public int getDropSlot() {
        return dropSlot;
    }

    public Stage getStage() {
        return stage;
    }

    public void advance(){
        if (stage == Stage.PICKUP){
            stage = Stage.DROPOFF;
        } else if (stage == Stage.DROPOFF){
            stage = Stage.DONE;
        }
    }

    public boolean isDone(){
        return stage == Stage.DONE;
    }



}
