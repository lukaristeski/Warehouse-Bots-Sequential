package Sequential;

import java.util.ArrayList;
import java.util.Locale;

public class GameState {

    public static int WIDTH;
    public static int HEIGHT;

    public static ArrayList<Bot> bots;
    public static ArrayList<Shelf> shelves;

    private static final int MAX_BOTS = 5;

    public static Bot getBot(int index) {
        return bots.get(index);
    }

    public static boolean isInside(int x, int y) {
        return x >= 0 && x < WIDTH && y >= 0 && y < HEIGHT;
    }

    public static boolean isWalkable(int x, int y) {
        if (!isInside(x, y)) return false;
        return !isOccupied(x,y);
    }

    public static boolean isOccupied(int x, int y) {

        if (shelves == null) return false;

        for (Shelf shelf : shelves) {
            Location loc = shelf.getLocation();
            if (loc.getX() == x && loc.getY() == y) {
                return true;
            }
        }

        for (Bot bot : bots) {
            Location loc = bot.getLocation();
            if (loc.getX() == x && loc.getY() == y) {
                return true;
            }
        }
        return false;
    }

    public static void initShelves(int shelfCols, int shelfRows) {
        shelves = new ArrayList<>();

        for (int sy = 0; sy < shelfRows; sy++){
            for (int sx = 0; sx < shelfCols; sx++){
                int tileX = sx * 2;
                int tileY = sy * 2;
                shelves.add(new Shelf(new Location(tileX, tileY)));
            }
        }

        }

    public static void spawnBotIfPossible(){
        if (bots.size() >= MAX_BOTS) return;

        if (!isOccupied(0,0)){
            bots.add(new Bot(new Location(0,2)));
        }
    }

    public static void update(){
        System.out.println("update");
        for (Bot bot : bots){

            if (bot.atTarget()){
                bot.finishCurrentTask();
                bot.clearTarget();
            }

            if (!bot.hasTarget() && bot.hasTasks()){
                Task t = bot.getCurrentTask();
                bot.setTarget(t.getPickupX(), t.getPickupY());
            }
            bot.moveTowardTarget();
        }
        spawnBotIfPossible();
    }

    public static Shelf getShelByShelfGrid(int shelfX, int shelfY){
        int tileX = shelfX * 2;
        int tileY = shelfY * 2;

        for (Shelf shelf : shelves) {
            Location loc = shelf.getLocation();
            if (loc.getX() == tileX && loc.getY() == tileY){
                return shelf;
            }
        }
        return null;
    }

    public static Location getInteractionTileForShelf(int shelfTileX, int shelfTileY){
        int[][] dirs = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

        for (int[] d : dirs){
            int nx = shelfTileX + d[0];
            int ny = shelfTileY + d[1];

            if (nx < 0 || ny < 0 || nx >= WIDTH || ny >= HEIGHT) continue;

            boolean blockedByShelf = false;
            for (Shelf s : shelves){
                Location l = s.getLocation();
                if (l.getX() == nx && l.getY() == ny){
                    blockedByShelf = true;
                    break;
                }
            }
            if (blockedByShelf) {
                return new Location(nx, ny);
            }
        }
        return null;
    }

    public static Location getInteractionTile(int shelfX, int shelfY){
        int tileX = shelfX * 2;
        int tileY = shelfY * 2;
        return getInteractionTileForShelf(tileX, tileY);
    }

    public static Location findFreeTile(){
        for (int y = 0; y < HEIGHT; y++) {
            for (int x = 0; x < WIDTH; x++) {
                if (!isOccupied(x, y)) return new Location(x, y);
            }
        }
        return new Location(0,0);
    }



}
