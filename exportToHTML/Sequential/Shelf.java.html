package Sequential;

public class Shelf{

    private Location location;
    private boolean[][] slots;

    public Shelf(Location location){
        this.location = location;
        slots = new boolean[2][8];

        for(int r = 0; r < 2; r++){
            for(int c = 0; c < 8; c++){
                slots[r][c] = true;
            }
        }
    }

    public Location getLocation(){
        return location;
    }

    private int slotRow(int slot){
        return slot / 8;
    }

    private int slotCol(int slot){
        return slot % 8;
    }

    public boolean hasItem(int slot){
        int r = slotRow(slot);
        int c = slotCol(slot);
        return slots[r][c];
    }

    public boolean pick(int slot){
        int r = slotRow(slot);
        int c = slotCol(slot);

        if (!slots[r][c]){
            return false;
        }
        slots[r][c] = false;
        return true;
    }

    public boolean drop(int slot){
        int r = slotRow(slot);
        int c = slotCol(slot);

        if (slots[r][c]){
            return false;
        }
        slots[r][c] = true;
        return true;
    }


}
